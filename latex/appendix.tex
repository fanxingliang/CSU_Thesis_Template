	function f=det2poly(P)
	n=size(P,4);
	f=0;
	if n==2  
	f=conv(P(:,:,1,1),P(:,:,2,2))-conv(P(:,:,1,2),P(:,:,2,1));  
	else
	P1=P;
	for i=1:n    
	P=P1;     
	P(:,:,1,:)=[];
	P(:,:,:,i)=[];   
	f=(-1)^(1+i)*conv(P1(:,:,1,i),det2poly(P))+f;   
	end
	end
\end{lstlisting}
